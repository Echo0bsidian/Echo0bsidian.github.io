Sub KeywordInjector()
  Dim keywords As String
  keywords = InputBox("Enter keywords separated by commas:")
  With ActiveDocument.Sections(1).Footers(wdFooterPrimary).Range
    .Font.Color = wdColorWhite
    .Text = keywords
  End With
End Sub
